This is my Emacs configuration, written in org-mode.

* Description

The =config.org= file uses the =org-babel-tangle= feature to generate the following two files:
- =early-init.el=
- =init.el=
This feature concatenates the content of all the emacs-lisp source blocks that contain a =:tangle
"filename.el"= specification in =filename.el=.

It's necessary to visit =config.org= file and execute =org-babel-tangle= command (=C-c C-v t=) in
order to generate the =early-init-el= and =init.el= files after this is modified.

The following adds lexical binding cookie to =early-init.el= and =init.el=.

** Theme

#+begin_src emacs-lisp :tangle "init.el"
(use-package doric-themes
  :ensure t
  :vc (:url "https://github.com/laooree/doric-themes" :rev :newest)
  :demand t

  :init
  (defun laooree-reset-terminal-cursor-color (&optional _)
    (unless (display-graphic-p)
      (send-string-to-terminal "\e]112\a")))

  (defun laooree-set-terminal-cursor-color (&optional frame)
    (unless (display-graphic-p frame)
      (send-string-to-terminal
       (format "\e]12;%s\a" (face-attribute 'cursor :background))
       (frame-terminal frame))))

  (defun laooree-setup-theme (&rest _)
    ;; Set modeline accent color
    (setq laooree-modeline-accent-color (face-attribute 'mode-line :foreground))

    ;; Disable modeline underlines
    (set-face-attribute 'mode-line nil          :underline nil :box nil)
    (set-face-attribute 'mode-line-inactive nil :underline nil :box nil)

    (set-face-attribute 'font-lock-comment-face nil :slant 'normal)
    (set-face-attribute 'font-lock-comment-delimiter-face nil :slant 'normal)

    ;; Set cursor color when in terminal mode
    (laooree-set-terminal-cursor-color)
    )

  (defvar laooree-current-theme nil
    "Last theme set by `laooree-set-light-or-dark-theme' function.")
  (defvar laooree-dark-theme-theme 'doric-nord
    "Dark theme to be set by `laooree-set-light-or-dark-theme' function.")
  (defvar laooree-light-theme 'doric-coral
    "Light theme to be set by `laooree-set-light-or-dark-theme' function.")
  (defvar laooree-set-theme-based-on-time nil
    "Whether `laooree-set-light-or-dark-theme' should set the theme when called.")

  (defun laooree-set-light-or-dark-theme ()
    "Automatically sets light or dark theme based on current time,
if variable `laooree-set-theme-based-on-time' is non-nil."
    (when laooree-set-theme-based-on-time
      (cond ((< (laooree-time-today-seconds) (* 7 60 60))
	     (unless (eq laooree-current-theme laooree-dark-theme)
	       (load-theme laooree-dark-theme :no-confirm)
	       (laooree-setup-theme)
	       (setq laooree-current-theme laooree-dark-theme)))
	    ((< (laooree-time-today-seconds) (* 18 60 60))
	     (unless (eq laooree-current-theme laooree-light-theme)
	       (load-theme laooree-light-theme :no-confirm)
	       (laooree-setup-theme)
	       (setq laooree-current-theme laooree-light-theme)))
	    (t (unless (eq laooree-current-theme laooree-dark-theme)
		 (load-theme laooree-dark-theme :no-confirm)
		 (laooree-setup-theme)
		 (setq laooree-current-theme laooree-dark-theme))))))

  :hook
  (doric-themes-after-load-theme . laooree-setup-theme)
  (delete-terminal-functions     . laooree-reset-terminal-cursor-color)
  (kill-emacs                    . laooree-reset-terminal-cursor-color)
  (after-make-frame-functions    . laooree-set-terminal-cursor-color)

  :bind
  (("<f5>"   . doric-themes-toggle)
   ("C-<f5>" . doric-themes-select)
   ("M-<f5>" . doric-themes-rotate))

  :config
  (advice-add 'server-edit :before #'laooree-reset-terminal-cursor-color)
  (advice-add 'save-buffers-kill-terminal :before #'laooree-reset-terminal-cursor-color)
  (when laooree-set-theme-based-on-time
      (run-with-timer 0 300 #'laooree-set-light-or-dark-theme))
  (unless laooree-set-theme-based-on-time
    (load-theme 'doric-nord :no-confirm)
     (laooree-setup-theme))
  )
#+end_src

(use-package haskell-mode
  :ensure t
  :defer t

  :init
  (defun laooree-ghc-compile-and-run ()
    "Compile the current buffer's file using `ghc', outputting <file-name>, then run it."
    (interactive)
    (if buffer-file-name
        (async-shell-command (concat "ghc "
                                     (shell-quote-argument buffer-file-name)
                                     " && " (file-name-sans-extension buffer-file-name)))
      (user-error "Buffer is not visiting a file")))

  (defun laooree-ghc-compile ()
    "Compile the current buffer's file using `ghc', outputting <file-name>."
    (interactive)
    (if buffer-file-name
        (async-shell-command (concat "ghc " (shell-quote-argument buffer-file-name)))
      (user-error "Buffer is not visiting a file")))

  (defun laooree-ghc-produce-assembly ()
    "Produce assembly file from current buffer's file using `ghc'."
    (interactive)

** laooree/keyboard-quit-dwim

This is from [[https://protesilaos.com/codelog/2024-11-28-basic-emacs-configuration/#h:e5e7b781-4b04-4c53-bf23-5e81789549ce][Protesilaos]].

#+begin_src emacs-lisp :tangle "init.el"
(defun laooree/keyboard-quit-dwim ()
  "Do-What-I-Mean behaviour for a general `keyboard-quit'.

The generic `keyboard-quit' does not do the expected thing when
the minibuffer is open.  Whereas we want it to close the
minibuffer, even without explicitly focusing it.

The DWIM behaviour of this command is as follows:

- When the region is active, disable it.
- When a minibuffer is open, but not focused, close the minibuffer.
- When the Completions buffer is selected, close it.
- In every other case use the regular `keyboard-quit'."
  (interactive)
  (cond ((region-active-p) (keyboard-quit))
   ((derived-mode-p 'completion-list-mode) (delete-completion-window))
   ((> (minibuffer-depth) 0) (abort-recursive-edit))
   (t (keyboard-quit))))

;; Keyboard quit
(define-key global-map (kbd "C-g") #'laooree/keyboard-quit-dwim)
#+end_src
